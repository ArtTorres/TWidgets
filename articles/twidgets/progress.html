ProgressChar

Displays an animated character with a message.

Layout

Supports text align, margin and position.

Events

  • OnStateChanged. Raises when the assigned text changes.

Implementation

Initialize and mount as the example below.

Source

// Setup the TWidget
var progress = new ProgressChar("twidget_id");
progress.Text = "Start Message";

// Mount and play
TWidgetPlayer.Mount(progress);

// Execute 4 Cycles
for (int i = 0; i < 4;)
{
    progress.Text = $"Message {++i:00}";
    Thread.Sleep(500);
}

ProgressBar

Displays an animated bar.

Layout

Supports text align, margin, padding, border and position.

Events

  • OnStateChanged. Raises when the assigned text changes.

Implementation

Initialize and mount as the example below.

Source

// Setup the TWidget
var progress = new ProgressBar("twidget_id");

// Mount and play
TWidgetPlayer.Mount(progress);

// Execute 5 Cycles
for (int i = 0; i < 5; i++)
{
    progress.PerformStep();
    Thread.Sleep(200);
}

ProgressList

Displays a list of progress bars.

Layout

Supports text align, margin, padding, border and position.

Events

  • OnStateChanged. Raises when the assigned text changes.

Implementation

Initialize and mount as the example below.

Source

// Setup the TWidget
var list = new ProgressList("twidget_id");
list.Items.Add(new ProgressItem() { Text = "Item A" });
list.Items.Add(new ProgressItem() { Text = "Item B" });
list.Items.Add(new ProgressItem() { Text = "Item C" });

// Mount and play
TWidgetPlayer.Mount(progressList);

// Execute Random Cycles
do
{
    list.Items[random.Next(list.Items.Count)].PerformStep();
    Thread.Sleep(500);
} while (!AreTasksFinished());

// Validation Function
bool AreTasksFinished()
{
    return 100 == list.Items.Select(s => s.Value).Sum() / list.Items.Count;
}
